  <h1>Rock Paper Scissors</h1>
  <div>
    <button onclick="play('Rock')">Rock</button>
    <button onclick="play('Paper')">Paper</button>
    <button onclick="play('Scissors')">Scissors</button>
  </div>
  <p id="result">Make your move!</p>
  <p id="score">Score: You 0 - 0 Computer</p>

  <script>
    let userScore = 0, compScore = 0;

    function play(userChoice) {
      const choices = ["Rock", "Paper", "Scissors"];
      const compChoice = choices[Math.floor(Math.random() * 3)];

      let result;
      if (userChoice === compChoice) {
        result = "Draw! Computer chose " + compChoice;
      } else if (
        (userChoice === "Rock" && compChoice === "Scissors") ||
        (userChoice === "Paper" && compChoice === "Rock") ||
        (userChoice === "Scissors" && compChoice === "Paper")
      ) {
        userScore++;
        result = "You win! Computer chose " + compChoice;
      } else {
        compScore++;
        result = "You lose! Computer chose " + compChoice;
      }

      document.getElementById("result").innerText = result;
      document.getElementById("score").innerText =
        "Score: You " + userScore + " - " + compScore + " Computer";
    }
  </script>
